
$request = new \Jasper\ImportTask();
$request->update = true;

$metadata = $client->startImportTask($request, file_get_contents('import_data.zip'));

$continue = true;
while($continue) {
    $state = $client->getImportState($metadata['id']);
    if ($state['phase'] == 'finished') {
        $continue = false;
        echo "<br><br>Import complete!";
    } else {
        echo "Still importing...<br>";
        @ob_flush();
        @flush();
        sleep(10);
    }
}
